function determinDevice() {
    var deviceType = null;

    if(/windows/i.test(navigator.userAgent)) {
        deviceType = {type: 'pc', title: 'PC (Windows)', mobile:false};
    }
    if(/linux/i.test(navigator.userAgent)) {
        deviceType = {type: 'linux', title: 'Linux', mobile:false};
    }
    if(/windows phone/i.test(navigator.userAgent)) {
        deviceType = {type: 'windowsphone', title: 'Windows Phone', mobile:true};
    }
    if(/android/i.test(navigator.userAgent)) {
        deviceType = {type: 'android', title: 'Android Device', mobile:true};
    }
    if(/ipad/i.test(navigator.userAgent)) {
        deviceType = {type: 'ipad', title: 'Apple Ipad', mobile:true};
    }
    if(/iphone/i.test(navigator.userAgent)) {
        deviceType = {type: 'iphone', title: 'Apple Iphone', mobile:true};
    }    
    if(/ipod/i.test(navigator.userAgent)) {
        deviceType = {type: 'ipod', title: 'Apple Ipod', mobile:true};
    }
    if(/blackberry/i.test(navigator.userAgent)) {
        deviceType = {type: 'blackberry', title: 'Blackberry', mobile:true};
    }
    if(/macintosh/i.test(navigator.userAgent)) {
        deviceType = {type: 'mac', title: 'Mac', mobile:false};
    }


    return deviceType;
}

                                vivid(target).prop('checked', true);
                            }
                        }
                        break;
                    case "textarea":
                        if(!data.id) {
                            var target = vivid(document)
                                .find('textarea[name="'+data.name+'"]')
                                .val(data.value);
                        } else if(data.id) {
                            var target = vivid(document)
                                .find('textarea#'+data.id)
                                .val(data.value);
                        }
                        break;
                    case "select":

                        var target = vivid(document).find('select[name="'+data.name+'"]')
                        var children = (target && target.length) ? vivid(target).children() : null;
                        if(!children) return false;

                        _.each(children, function(child){
                            if(vivid(child).val() !== data.value) {
                                vivid(child).prop('selected', false);
                            } else if(vivid(child).val() === data.value) {
                                vivid(child).prop('selected', true);
                            }
                        })
                        break;
                    case "windowScroll":
                            vivid(window).scrollTop(data.value);
                        break;
                    case "clearFormElements":
                        var formsElements = vivid(document)
                            .find('input, textarea');
                        if(formsElements && formsElements.length) {
                            _.each(formsElements, function(el){
                                vivid(el).val('');
                            })
                        }
                        break;
                    default:
                        return false;
                }
                
            })

            window.socket.on('reconnect', function(){
                return window.location.reload();
            })

            vivid(document).on('click', 'a', function(e){
                e.preventDefault();
                e.stopPropagation();
                App.isAnchor = true;
                var href = $(e.target).attr('href');
                socket.emit('navigateWindow', App.site+href);
            })


            function emitEvent(event) {
                event.preventDefault();
                var data = $(event.currentTarget).val();
                var toSubmit = {
                    id: $(event.currentTarget).attr('id')
                    , name: $(event.currentTarget).attr('name')
                    , value: data
                    , type: $(event.target).get(0).tagName.toLowerCase()
                    , tagType: $(event.target).attr('type')
                };            
                socket.emit('manipulateFrame', toSubmit);
            }

            vivid(document).on('keyup', 'input, textarea', function(event){
                emitEvent(event);
            })

            vivid(document).on('focusout', 'input[type="text"], textarea', function(event){
                emitEvent(event);
            })

            vivid(document).on('change', 'input[type="radio"],input[type="checkbox"]', function(event){
                emitEvent(event);
            })

            vivid(document)
                .find('input, textarea')
                .each(function(field){
                    $(field).val('');
                })

            var lastScrollTimeout = null;

            vivid(window).on('scroll', function(event){
                if(App.device && App.device.mobile) return false;

                if(lastScrollTimeout) {
                    clearTimeout(lastScrollTimeout);
                }
                lastScrollTimeout = setTimeout(function(){
                    var scrolledTop = $(event.target).scrollTop();
                    socket.emit('manipulateFrame', {type: 'windowScroll', value: scrolledTop});
                }, 500);
            })

            vivid(window).on('beforeunload', function(event){
                if(!App.isAnchor) {
                    socket.emit('reloadSite');
                }
            })
            vivid(document).ready(function(){
                App.device = determinDevice();
                var styleSheets = vivid(document).find('[rel="stylesheet"]');
                $(styleSheets).each(function(i, stylesheet){
                    App.styleSheets.push($(stylesheet).attr('href'));
                })
            })

            
            
        })(jQuery)
